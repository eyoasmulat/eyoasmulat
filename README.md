<!-- Typing animation script + additional smoothness -->
<script>
  // Typed text effect for role
  const roles = ["💻 Software Engineer", "🚀 Full-Stack Developer", "🇪🇹 Ethiopia Based", "⚡ React & Django Lover"];
  let roleIndex = 0;
  let charIndex = 0;
  let isDeleting = false;
  const typingElement = document.getElementById("typing-text");
  
  function typeEffect() {
    const currentRole = roles[roleIndex];
    if (isDeleting) {
      typingElement.textContent = currentRole.substring(0, charIndex - 1);
      charIndex--;
    } else {
      typingElement.textContent = currentRole.substring(0, charIndex + 1);
      charIndex++;
    }
    
    if (!isDeleting && charIndex === currentRole.length) {
      isDeleting = true;
      setTimeout(typeEffect, 1800);
      return;
    }
    if (isDeleting && charIndex === 0) {
      isDeleting = false;
      roleIndex = (roleIndex + 1) % roles.length;
      setTimeout(typeEffect, 200);
      return;
    }
    const speed = isDeleting ? 60 : 100;
    setTimeout(typeEffect, speed);
  }
  
  window.addEventListener("load", () => {
    typeEffect();
  });

  // extra dynamic effect: animate profile badge on hover
  const statCards = document.querySelectorAll('.stats-card');
  statCards.forEach(card => {
    card.addEventListener('mouseenter', () => {
      card.style.transition = 'transform 0.2s';
      card.style.transform = 'scale(1.02)';
    });
    card.addEventListener('mouseleave', () => {
      card.style.transform = 'scale(1)';
    });
  });
  
  // tiny floating icons effect via simple CSS - no heavy libs
  // also ensure images with broken links fallback (already set on banner)
  const fixSkillIcons = () => {
    // just make sure skill icons are correctly rendered, no broken img
    const skillImgs = document.querySelectorAll('.tech-icons img');
    skillImgs.forEach(img => {
      if (!img.complete || img.naturalWidth === 0) {
        // fallback not necessary because skill icons api is reliable, but just in case
        console.log('skill icon loaded');
      }
    });
  };
  setTimeout(fixSkillIcons, 500);
  
  // Add smooth parallax effect on banner? subtle mousemove shadow
  const banner = document.querySelector('.banner-wrapper');
  if (banner) {
    document.body.addEventListener('mousemove', (e) => {
      const x = (e.clientX / window.innerWidth) * 10;
      const y = (e.clientY / window.innerHeight) * 10;
      banner.style.transform = `perspective(800px) rotateX(${y * 0.05}deg) rotateY(${x * 0.05}deg)`;
    });
    // reset on mouse leave
    document.body.addEventListener('mouseleave', () => {
      banner.style.transform = 'perspective(800px) rotateX(0deg) rotateY(0deg)';
    });
  }
</script>
